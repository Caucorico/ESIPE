mk :: a -> Var a Source #

mk n makes a propositional variable with name n.

>>> [mk i | i <- [1..4]]
[1,2,3,4]
>>> [mk i | i <- ['a'..'d']]
['a','b','c','d']
>>> [mk o | o <- [LT .. GT]]
[LT,EQ,GT]
